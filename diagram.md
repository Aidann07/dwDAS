@startuml
actor Customer as C
participant "Жүйе" as S
participant "Төлем шлюзі" as PG
participant "Администратор" as A
participant "Подрядчик 1" as P1
participant "Подрядчик 2" as P2
participant "Менеджер" as M

C -> S: Площадка қолжетімді ме?
S -> S: Тексеру
alt қолжетімді
S --> C: Ақпарат беру + баға
else қолжетімсіз
S --> C: Басқа күн таңдаңыз
return
end

== Төлем ==
C -> S: Брондауды растау
S -> PG: Төлем сұранысы
alt төлем сәтті
PG --> S: OK
S -> A: Брон расталды\n(хабарлама)
else төлем сәтсіз
PG --> S: Қате
S -> C: Төлемді қайта жасаңыз
return
end

== Ұйымдастыру ==
A -> A: Тапсырмалар жоспарлау
par Подрядчиктерге хабарлау
A -> P1: Тапсырма
A -> P2: Тапсырма
end
par Тапсырманы орындау
P1 -> A: Аяқталды
P2 -> A: Аяқталды
end
A -> S: Барлығы дайын

== Мероприятие аяқталуы ==
S -> C: Пікір қалдыру
S -> M: Пікірлер туралы есеп

@enduml
